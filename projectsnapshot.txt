const express = require('express');
const path = require('path');
const dotenv = require('dotenv');
const connectDB = require('./config/db');
const session = require('express-session');
const MongoStore = require('connect-mongo');

dotenv.config();
connectDB();

const app = express();

app.set('views', path.join(__dirname, 'views'));
app.set('view engine', 'ejs');

app.use(express.urlencoded({ extended: true }));
app.use(express.json());
app.use(express.static(path.join(__dirname, 'public')));

app.use(session({
    secret: process.env.SESSION_SECRET || 'litera-secure-key',
    resave: false,
    saveUninitialized: false,
    store: MongoStore.create({ mongoUrl: process.env.MONGO_URI }),
    cookie: {
        maxAge: 1000 * 60 * 60 * 24
    }
}));

app.use((req, res, next) => {
    res.locals.user = req.session.user || null;
    res.locals.session = req.session;
    res.locals.cartCount = req.session.cart ? req.session.cart.reduce((sum, item) => sum + item.qty, 0) : 0;
    next();
});

app.get('/', (req, res) => {
    res.render('pages/landing', { 
        pageTitle: 'Welcome to Litera Ecosystem' 
    });
});

app.use('/auth', require('./routes/auth'));
app.use('/admin', require('./routes/admin'));
app.use('/', require('./routes/store'));

app.use((err, req, res, next) => {
    console.error(err.stack);
    res.status(500).send('Something went wrong on the Litera Server.');
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => console.log(`[Litera Server]: Running on http://localhost:${PORT}`));

const multer = require('multer');
const path = require('path');

const storage = multer.diskStorage({
    destination: (req, file, cb) => {
        cb(null, 'public/img/covers');
    },
    filename: (req, file, cb) => {
        const uniqueSuffix = Date.now() + '-' + Math.round(Math.random() * 1E9);
        cb(null, file.fieldname + '-' + uniqueSuffix + path.extname(file.originalname));
    }
});

const fileFilter = (req, file, cb) => {
    const allowedFileTypes = /jpeg|jpg|png|gif|webp/;
    const extname = allowedFileTypes.test(path.extname(file.originalname).toLowerCase());
    const mimetype = allowedFileTypes.test(file.mimetype);

    if (mimetype && extname) {
        return cb(null, true);
    } else {
        cb(new Error('Error: Only image files (jpeg, jpg, png, gif, webp) are allowed!'), false);
    }
};

const upload = multer({
    storage: storage,
    limits: {
        fileSize: 5 * 1024 * 1024
    },
    fileFilter: fileFilter
});

module.exports = upload.single('coverImage');

<script>
    document.addEventListener("DOMContentLoaded", () => {
        const qtyInputs = document.querySelectorAll('.checkout-qty-input');
        const subtotalDisplay = document.getElementById('checkoutSubtotalDisplay');
        const totalDisplay = document.getElementById('checkoutTotalDisplay');

        function recalculateCheckoutSummary() {
            let overallSubtotal = 0;
            
            document.querySelectorAll('.checkout-qty-input').forEach(input => {
                const currentQty = parseInt(input.value) || 1;
                const unitPrice = parseFloat(input.getAttribute('data-price')) || 0;
                const rowTotal = currentQty * unitPrice;
                
                const rowTotalNode = input.closest('.item-checkout-row').querySelector('.row-item-total');
                if (rowTotalNode) {
                    rowTotalNode.textContent = `Rp ${rowTotal.toLocaleString('id-ID')}`;
                }
                
                overallSubtotal += rowTotal;
            });

            if (subtotalDisplay) subtotalDisplay.textContent = `Rp ${overallSubtotal.toLocaleString('id-ID')}`;
            if (totalDisplay) totalDisplay.textContent = `Rp ${overallSubtotal.toLocaleString('id-ID')}`;
        }

        qtyInputs.forEach(input => {
            input.addEventListener('input', () => {
                if (parseInt(input.value) < 1 || !input.value) {
                    input.value = 1;
                }
                recalculateCheckoutSummary();
            });

            input.addEventListener('change', () => {
                recalculateCheckoutSummary();
            });
        });
    });
</script>

<script>
    function showToastAlert(textMessage) {
        const toastNode = document.getElementById('runtime-popup-alert');
        if (toastNode) {
            toastNode.innerHTML = `<i class="bi bi-check-circle-fill me-2"></i>${textMessage}`;
            toastNode.style.display = 'block';
            setTimeout(() => {
                toastNode.style.display = 'none';
            }, 3500);
        }
    }

    document.addEventListener("DOMContentLoaded", () => {
        const cartForms = document.querySelectorAll('.ajax-cart-form');
        
        cartForms.forEach(form => {
            form.addEventListener('submit', async (event) => {
                event.preventDefault();
                
                const targetUrl = form.getAttribute('action');
                const submitButton = form.querySelector('button[type="submit"]');
                const originalButtonHtml = submitButton.innerHTML;
                
                submitButton.disabled = true;
                submitButton.innerHTML = '<span class="spinner-border spinner-border-sm me-1" role="status" aria-hidden="true"></span> Adding...';
                
                try {
                    const response = await fetch(targetUrl, {
                        method: 'POST',
                        headers: {
                            'Content-Type': 'application/json',
                            'X-Requested-With': 'XMLHttpRequest'
                        }
                    });
                    
                    const result = await response.json();
                    if (response.ok && result.success) {
                        const cartBadge = document.querySelector('.navbar .badge');
                        if (cartBadge) {
                            cartBadge.textContent = result.cartCount;
                        }
                        showToastAlert(result.message);
                    } else {
                        alert(result.error || 'Failed to add item to cart.');
                    }
                } catch (error) {
                    console.error('AJAX Cart operational failure:', error);
                    alert('Network error. Failed to communicate with server.');
                } finally {
                    submitButton.disabled = false;
                    submitButton.innerHTML = originalButtonHtml;
                }
            });
        });
    });
</script>

                    event.preventDefault();

                        alert('Network error. Failed to communicate with server.');
                    } finally {
                        submitButton.disabled = false;
                        submitButton.innerHTML = originalButtonHtml;
                    }
                });
            });
        });

<script>
    document.addEventListener('DOMContentLoaded', () => {
        const starElements = document.querySelectorAll('#star-rating .star-icon');
        const ratingInputNode = document.getElementById('rating-input');
        const validationFeedbackNode = document.getElementById('ratingValidationFeedback');
        const formEngine = document.getElementById('publicAssetReviewForm');

        starElements.forEach(star => {
            star.addEventListener('mouseover', function() {
                const currentHoverValue = parseInt(this.getAttribute('data-value'));
                highlightStarsMatrix(currentHoverValue, 'hover');
            });

            star.addEventListener('mouseout', function() {
                removeStarsStyleClass('hover');
            });

            star.addEventListener('click', function() {
                const selectedRatingValue = parseInt(this.getAttribute('data-value'));
                ratingInputNode.value = selectedRatingValue;
                
                removeStarsStyleClass('selected');
                highlightStarsMatrix(selectedRatingValue, 'selected');
                
                if (selectedRatingValue > 0 && validationFeedbackNode) {
                    validationFeedbackNode.classList.add('d-none');
                }
            });
        });

        function highlightStarsMatrix(limitValue, className) {
            starElements.forEach(star => {
                const starValue = parseInt(star.getAttribute('data-value'));
                if (starValue <= limitValue) {
                    star.classList.add(className);
                    if (className === 'selected') {
                        star.classList.remove('text-muted');
                    }
                }
            });
        }

        function removeStarsStyleClass(className) {
            starElements.forEach(star => {
                star.classList.remove(className);
                if (className === 'selected') {
                    star.classList.add('text-muted');
                }
            });
        }

        formEngine.addEventListener('submit', function(event) {
            const finalRatingScore = parseInt(ratingInputNode.value);
            
            if (finalRatingScore === 0 || isNaN(finalRatingScore)) {
                event.preventDefault(); 
                if (validationFeedbackNode) {
                    validationFeedbackNode.classList.remove('d-none');
                }
            }
        });
    });
</script>

<script>
    document.addEventListener('DOMContentLoaded', function() {
        const structuralTabTriggers = document.querySelectorAll('button[data-bs-toggle="tab"]');

        structuralTabTriggers.forEach(tab => {
            tab.addEventListener('shown.bs.tab', function (event) {
                const targetedContainerId = event.target.getAttribute('data-bs-target');
                
                let operationalDataUrl = '';
                let bindingContentId = '';

                if (targetedContainerId === '#transaction-ledger') {
                    operationalDataUrl = '/admin/data/transactions'; 
                    bindingContentId = 'transactions-content-container';
                } else if (targetedContainerId === '#review-logs') {
                    operationalDataUrl = '/admin/data/reviews';
                    bindingContentId = 'reviews-content-container';
                } else {
                    return;
                }

                executeDynamicPipelineFetch(operationalDataUrl, bindingContentId, targetedContainerId);
            });
        });

        async function executeDynamicPipelineFetch(url, contentFieldId, identifierTag) {
            const outputDomContainer = document.getElementById(contentFieldId);
            outputDomContainer.innerHTML = `
                <div class="d-flex justify-content-center align-items-center p-5">
                    <div class="spinner-border text-primary me-3" role="status"></div>
                    <span class="fw-medium text-secondary">Fetching records from server node...</span>
                </div>
            `;
            try {
                const operationalResponse = await fetch(url);
                if (!operationalResponse.ok) throw new Error(`HTTP Error Status: ${operationalResponse.status}`);
                
                const responsePayload = await operationalResponse.json();
                if (responsePayload.success) {
                    let structuredHtmlOutput = '';
                    if (identifierTag === '#transaction-ledger') {
                        structuredHtmlOutput = constructTransactionsTableStructure(responsePayload.transactions);
                    } else if (identifierTag === '#review-logs') {
                        structuredHtmlOutput = constructReviewsTableStructure(responsePayload.reviews);
                    }
                    outputDomContainer.innerHTML = structuredHtmlOutput;
                } else {
                    outputDomContainer.innerHTML = `<div class="alert alert-danger mb-0">${responsePayload.message || 'Data parsing execution failure.'}</div>`;
                }
            } catch (runtimeError) {
                outputDomContainer.innerHTML = `<div class="alert alert-danger mb-0">Connection pipeline failure: ${runtimeError.message}</div>`;
            }
        }

        window.confirmOrderAction = async function(orderId) {
            if (!confirm("Mark this order as Completed? This confirms payment settlement and product hand-over.")) return;
            try {
                const response = await fetch(`/admin/orders/complete/${orderId}`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' }
                });
                const data = await response.json();
                if (response.ok && data.success) {
                    alert("Order status successfully updated to Completed!");
                    executeDynamicPipelineFetch('/admin/data/transactions', 'transactions-content-container', '#transaction-ledger');
                } else {
                    alert(data.message || "Failed to update order status.");
                }
            } catch (err) {
                console.error(err);
                alert("Network communication error.");
            }
        }

        window.deleteReviewAction = async function(reviewId) {
            if (!confirm("Are you sure you want to permanently delete this customer review? This action cannot be undone.")) return;
            try {
                const response = await fetch(`/admin/reviews/delete/${reviewId}`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' }
                });
                const data = await response.json();
                if (response.ok && data.success) {
                    alert("Customer review successfully deleted!");
                    executeDynamicPipelineFetch('/admin/data/reviews', 'reviews-content-container', '#review-logs');
                } else {
                    alert(data.message || "Failed to delete review.");
                }
            } catch (err) {
                console.error(err);
                alert("Network error processing deletion command.");
            }
        }

        function constructTransactionsTableStructure(transactions) {
            if (!transactions || transactions.length === 0) {
                return '<div class="alert alert-warning mb-0 text-center">No order history records found within the store database.</div>';
            }

            let structuredTable = `
                <div class="table-responsive" style="max-height: 500px; overflow-y: auto;">
                    <table class="table table-striped table-hover align-middle table-sm mb-0">
                        <thead class="table-dark sticky-top">
                            <tr>
                                <th scope="col">Order ID</th>
                                <th scope="col">Customer</th>
                                <th scope="col">Total Price</th>
                                <th scope="col">Payment Method</th>
                                <th scope="col">Status</th>
                                <th scope="col">Order Date</th>
                                <th scope="col" class="text-center">Control Actions</th>
                            </tr>
                        </thead>
                        <tbody>
            `;

            transactions.forEach(tx => {
                const currentStatus = tx.status || 'Pending Pickup';
                let conditionalStatusBadge = 'bg-warning text-dark';
                
                if (currentStatus === 'Completed') conditionalStatusBadge = 'bg-success';
                else if (currentStatus === 'Cancelled' || currentStatus === 'Expired') conditionalStatusBadge = 'bg-danger';

                const localizedDateOutput = new Date(tx.createdAt).toLocaleDateString('id-ID', { year: 'numeric', month: 'short', day: 'numeric' });

                let actionButtonHtml = '';
                if (currentStatus !== 'Completed' && currentStatus !== 'Cancelled') {
                    actionButtonHtml = `
                        <button onclick="confirmOrderAction('${tx._id}')" class="btn btn-xs btn-outline-success fw-bold rounded-pill px-2" style="font-size:0.7rem;">
                            <i class="bi bi-check2-circle me-1"></i>Mark Complete
                        </button>
                    `;
                } else {
                    actionButtonHtml = `<span class="text-muted small fst-italic">No action required</span>`;
                }

                structuredTable += `
                    <tr>
                        <td><small class="text-muted font-monospace">${tx._id}</small></td>
                        <td class="fw-semibold text-dark">${tx.user ? tx.user.username : 'Account Deleted'}</td>
                        <td class="text-danger fw-bold">Rp ${tx.totalPrice.toLocaleString('id-ID')}</td>
                        <td><span class="small font-monospace fw-semibold">${tx.paymentMethod}</span></td>
                        <td><span class="badge ${conditionalStatusBadge} rounded-pill px-2.5">${currentStatus}</span></td>
                        <td class="text-muted small">${localizedDateOutput}</td>
                        <td class="text-center">${actionButtonHtml}</td>
                    </tr>
                `;
            });
            structuredTable += `</tbody></table></div>`;
            return structuredTable;
        }

        function constructReviewsTableStructure(reviews) {
            if (!reviews || reviews.length === 0) {
                return '<div class="alert alert-warning mb-0 text-center">No customer feedback reviews have been logged for any book yet.</div>';
            }

            let structuredTable = `
                <div class="table-responsive" style="max-height: 500px; overflow-y: auto;">
                    <table class="table table-striped table-hover align-middle table-sm mb-0">
                        <thead class="table-dark sticky-top">
                            <tr>
                                <th scope="col" style="width: 25%;">Target Book</th>
                                <th scope="col" style="width: 20%;">Reviewer Name</th>
                                <th scope="col" style="width: 15%;">Rating Stars</th>
                                <th scope="col" style="width: 30%;">Comment</th>
                                <th scope="col" style="width: 10%;" class="text-center">Action</th>
                            </tr>
                        </thead>
                        <tbody>
            `;
            reviews.forEach(review => {
                const numericalScore = parseInt(review.rating) || 0;
                const dynamicStarsMatrix = '★'.repeat(numericalScore) + '☆'.repeat(5 - numericalScore);
                const truncatedTextBody = review.comment.length > 80 ? review.comment.substring(0, 80) + '...' : review.comment;
                
                structuredTable += `
                    <tr>
                        <td><strong class="text-dark">${review.book ? review.book.title : 'Book Removed'}</strong></td>
                        <td class="fw-medium text-secondary">${review.user ? review.user.username : 'Anonymous'}</td>
                        <td><span class="text-warning font-monospace">${dynamicStarsMatrix}</span></td>
                        <td class="text-muted fst-italic">"${truncatedTextBody}"</td>
                        <td class="text-center">
                            <button onclick="deleteReviewAction('${review._id}')" class="btn btn-sm btn-outline-danger rounded-pill fw-bold" style="font-size:0.7rem; padding: 2px 10px;">
                                <i class="bi bi-trash3-fill"></i>
                            </button>
                        </td>
                    </tr>
                `;
            });
            structuredTable += `</tbody></table></div>`;
            return structuredTable;
        }
    });

<script>
    const interactiveRatingStars = document.querySelectorAll(".review-rating-star-node");
    const hiddenRatingPayloadInput = document.getElementById("reviewRatingSecurePayload");

    if (interactiveRatingStars && interactiveRatingStars.length > 0) {
        interactiveRatingStars.forEach(star => {
            star.addEventListener("click", () => {
                const targetRatingValue = parseInt(star.getAttribute("data-value")) || 0;
                if (hiddenRatingPayloadInput) {
                    hiddenRatingPayloadInput.value = targetRatingValue;
                }
                interactiveRatingStars.forEach(s => s.classList.remove("selected"));
                for (let i = 0; i < targetRatingValue; i++) {
                    interactiveRatingStars[i].classList.add("selected");
                }
            });

            star.addEventListener("mouseover", () => {
                const hoverRatingThreshold = parseInt(star.getAttribute("data-value")) || 0;
                interactiveRatingStars.forEach(s => s.classList.remove("hover"));
                for (let i = 0; i < hoverRatingThreshold; i++) {
                    interactiveRatingStars[i].classList.add("hover");
                }
            });
            star.addEventListener("mouseleave", () => {
                interactiveRatingStars.forEach(s => s.classList.remove("hover"));
            });
        });
    }
</script>
